### 创建多线程
* 普通方式
```py
import threading
import time


def coding():
    for x in range(3):
        print("正在 coding %s" % x)
        time.sleep(1)  

def painting():
    for x in range(3):
        print(x)
        time.sleep(1)

def main():
    t1 = threading.Thread(target=coding)
    t2 = threading.Thread(target=painting)
    t1.start()
    t2.start()

main()
```
* 通过类的方式
```py
class CodingThread(threading.Thread):
    def run(self):
        for x in range(3):
            print("正在 coding %s" % threading.current_thread())

class DrawingThread(threading.Thread):
    def run(self):
        for x in range(3):
            print("正在 painting %s" % threading.current_thread())


def main():
    t1 = CodingThread()
    t2 = DrawingThread()

    t1.start()
    t2.start()
```

### 多线程共享全局变量
```py
# 定义在外部的全局变量
value = 0

def addValue():
    # 用到的全局变量, 要先声明
    global value

    print(value)
    for x in range(1000):
        value += 1
    print(value)
```

* 添加锁, 保证数据整齐
```py
value = 0  # 全局变量
gLock = threading.Lock()  # 定义锁

def addValue():
    global value  # 声明在局部空间使用全局变量
    
    gLock.acquire()  # 加锁
    for x in range(100000):
        value += 1
    gLock.release()  # 解锁
    print(value)  # 查看值


for x in range(3):
    t = threading.Thread(target=addValue)
    t.start()
```
